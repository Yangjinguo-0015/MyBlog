    # 循环控制语句

    # for循环

    # for循环由三部分组成

    # var i=0 变量初始化 i<arr.length循环执行的条件，一旦不符合条件，就会退出循环，i++步长，每次加1

    # 常用于一些已知结束条件的循环

      var arr=[1,2,3,4,5];
  for(var i=0;i<arr.length;i++){
    console.log(arr[i]) // 1 2 3 4 5
  }

    # while 循环

    # 基本上所有的循环都是由三部分组成,i=0初始化变量,i<len循环执行的条件,i++ 步长,每次加1

    # 常用于一些未知循环条件的场景

      var arr=[1,2,3,4,5],len=arr.length,i=0;
  while(i<len){
    console.log(arr[i])//1 2 3 4 5
    i++;
  }

    # do while 循环

    # do while和while的区别就是do while循环会先执行再判断是否达到终止条件，所以它会多执行一次

      var arr=[1,2,3,4,5],len=arr.length,i=0;
  do{
    console.log(arr[i])//1 2 3 
    i++
  }while(i<3)

    # if else

    # if 判断条件为真时做的一些事情，其他的就在else的代码块里面去写

      var time=20
  if(time<20){
    //do something
  }else{

  }

    # if else if else if和else中间可以有很多个else if判断

      var time=20
  if(time<20){
    //do something
  }else if(time==20){

  }else{

  }

    # switch语句

    # 当有三条或者三条以上else if 时，推荐使用switch语句

    # break退出循环 countine 跳出当前循环，进入下一次循环

      var time=20
  switch(time){
     case10:dosomething;break;
     case20:dosomething;break;
     case30:dosomething;break;
     case40:dosomething;break;
     default:dosomething;break;
  }
